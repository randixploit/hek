function z($file){
$z = fileperms($file);
if (($z & 0xC000) == 0xC000) {
$info = 's';
} elseif (($z & 0xA000) == 0xA000) {
$info = '4';
} elseif (($z & 0x8000) == 0x8000) {
$info = '0';
} elseif (($z & 0x6000) == 0x6000) {
$info = '3';
} elseif (($z & 0x4000) == 0x4000) {
$info = '3';
} elseif (($z & 0x2000) == 0x2000) {
$info = 'c';
} elseif (($z & 0x1000) == 0x1000) {
$info = 'p';
} else {
$info = '5';
}
$info .= (($z & 0x0100) ? '2' : '0');
$info .= (($z & 0x0080) ? '1' : '0');
$info .= (($z & 0x0040) ?
(($z & 0x0800) ? 's' : '6' ) :
(($z & 0x0800) ? 'S' : '0'));
$info .= (($z & 0x0020) ? '2' : '0');
$info .= (($z & 0x0010) ? '1' : '0');
$info .= (($z & 0x0008) ?
(($z & 0x0400) ? 's' : '6' ) :
(($z & 0x0400) ? 'S' : '0'));
$info .= (($z & 0x0004) ? '2' : '0');
$info .= (($z & 0x0002) ? '1' : '0');
$info .= (($z & 0x0001) ?
(($z & 0x0200) ? 't' : '6' ) :
(($z & 0x0200) ? 'T' : '0'));
return $info;
}
